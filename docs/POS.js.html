/** @module koalanlp/POS **/

/** [체언] ''일반명사'': 일반 개념을 표시하는 명사. **/
export const NNG = "NNG";
/** [체언] ''고유명사'' : 낱낱의 특정한 사물이나 사람을 다른 것들과 구별하여 부르기 위하여 고유의 기호를 붙인 이름. **/
export const NNP = "NNP";
/** [체언] ''일반 의존명사'' : 의미가 형식적이어서 다른 말 아래에 기대어 쓰이는 명사.
 * ‘것1’, ‘따름1’, ‘뿐1’, ‘데1’ 따위가 있다.
 * */
export const NNB = "NNB";
/** [체언] ''단위성 의존명사'' : 수효나 분량 따위의 단위를 나타내는 의존 명사.
 * ‘쌀 한 말, 쇠고기 한 근, 굴비 한 두름, 북어 한 쾌, 고무신 한 켤레, 광목 한 필’에서 ‘말3’, ‘근3’, ‘두름1’, ‘쾌1’, ‘켤레2’, ‘필2’
 * */
export const NNM = "NNM";
/** [체언] ''수사'' : 사물의 수량이나 순서를 나타내는 품사. 양수사와 서수사가 있다. **/
export const NR = "NR";
/** [체언] ''대명사'' : 사람이나 사물의 이름을 대신 나타내는 말. 또는 그런 말들을 지칭하는 품사 **/
export const NP = "NP";

export const VA = "VA";
/** [용언] ''보조용언'' : 본용언과 연결되어 그것의 뜻을 보충하는 역할을 하는 용언. 보조 동사, 보조 형용사가 있다.
 * ‘ 가지고 싶다’의 ‘싶다’, ‘먹어 보다’의 ‘보다1’ 따위이다.
 * */
export const VX = "VX";
/** [용언] ''긍정지정사(이다)'': 무엇이 무엇이라고 지정하는 단어. (이다) **/
export const VCP = "VCP";

export const VCN = "VCN";

/** [수식언] ''관형사'' : 체언 앞에 놓여서, 그 체언의 내용을 자세히 꾸며 주는 품사. **/
export const MM = "MM";

export const IC = "IC";

const _NOUN_SET = [NNG, NNP, NNB, NNM, NR, NP];
const _PRED_SET = [VV, VA, VX, VCP, VCN];
const _MODF_SET = [MM, MAG, MAJ];
const _JOSA_SET = [JKS, JKC, JKG, JKO, JKB, JKV, JKQ, JC, JX];
const _EOMI_SET = [EP, EF, EC, ETN, ETM];
const _AFFX_SET = [XPN, XPV, XSN, XSV, XSM, XSA, XSO, XR];
const _SUFX_SET = [XSN, XSV, XSM, XSA, XSO];
const _SYMB_SET = [SF, SP, SS, SE, SW, SO];
const _UNKN_SET = [NF, NV, NA];

let finder = function(set, tag){
    if(typeof tag === "string")
        return set.indexOf(tag) >= 0;
    else if(Object.keys(tag).indexOf("tag") >= 0)
        return set.indexOf(tag.tag) >= 0;
    else return false;
};

/**
 * 주어진 tag가 체언(명사,대명사,의존명사,수사)인지 확인.
 * @function
 * @param {string|Morpheme} tag 확인할 품사표기 또는 형태소.
 */
export let isNoun = function(tag){
    return finder(_NOUN_SET, tag);
};
